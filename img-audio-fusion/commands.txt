## 1) 生成图像-音频对齐划分与标签（8:1:1）
source /root/miniconda3/etc/profile.d/conda.sh && conda activate lsmrt_py311 && python /root/mqx/LSMRT/img-audio-fusion/prepare_img_audio_split.py --img_dir data/fusiondata/img --audio_dir data/fusiondata/audio --img_runtime_dir data/fusiondata/img_jpg --out_dir img-audio-fusion/split_fusiondata_8_1_1 --train_ratio 0.8 --val_ratio 0.1 --test_ratio 0.1 --seed 42

## 2) 双卡 GPU 训练融合头（冻结图像/音频预训练分支）
source /root/miniconda3/etc/profile.d/conda.sh && conda activate lsmrt_py311 && CUDA_VISIBLE_DEVICES=0,1 python /root/mqx/LSMRT/img-audio-fusion/train_fusion_head.py --split_dir img-audio-fusion/split_fusiondata_8_1_1 --image_ckpt /root/mqx/LSMRT/image-branch/runs/effb0_s224_split1_resize/best.pt --audio_ckpt /root/mqx/LSMRT/audio-branch/runs/horizontal_robot_impact_sr24000_clip1.0_b32x2_e50_pretrain_invSqrt_c5/best.pt --out_dir img-audio-fusion/runs/fusion_head_split811_seed42 --img_size 224 --target_sr 24000 --clip_sec 1.0 --epochs 150 --batch 64 --lr 1e-3 --wd 1e-4 --dropout 0.3 --hidden_dim 512 --label_smoothing 0.05 --patience 25 --num_workers 4 --seed 42 --use_cuda --use_data_parallel
